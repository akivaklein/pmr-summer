function runDataTests() {
  if (typeof console === 'undefined') return;

  console.assert(Array.isArray(cabins), 'cabins should be an array');
  console.assert(
    cabins.length === 10,
    'cabins should match the 10 cabin sections currently shown on the PMR cabins page'
  );
  console.assert(filters.includes('All'), 'filters should include All');
  console.assert(
    categoryFilters.includes('Cabins'),
    'category filters should include Cabins'
  );
  console.assert(
    cabins.every((cabin) => cabin.name && cabin.category && cabin.cta),
    'every cabin needs name, category, and cta'
  );
  console.assert(
    cabins.every((cabin) => filters.includes(cabin.filter)),
    'every cabin filter should exist in filters'
  );
  console.assert(
    cabins.every((cabin) => categoryFilters.includes(cabin.category)),
    'every category should exist in categoryFilters'
  );
  console.assert(
    cabins.every((cabin) =>
      cabin.image.includes('presidentialmountainresort.com/wp-content/uploads')
    ),
    'all cabin images should use PMR website images, not placeholder stock photos'
  );
  console.assert(
    cabins.some(
      (cabin) =>
        cabin.name === 'Oak Premier Cabin' && cabin.sleeps.includes('12')
    ),
    'Oak Premier should remain listed as up to 12 guests'
  );
  console.assert(
    cabins.some(
      (cabin) =>
        cabin.name === 'Rosewood Cottage' &&
        cabin.features.includes('Lake views')
    ),
    'Rosewood Cottage should keep lake-view positioning'
  );
  console.assert(
    mapHotspots.length >= 6,
    'interactive map should include multiple clickable resort areas'
  );
  console.assert(
    mapHotspots.every((spot) =>
      cabins.some((cabin) => cabin.name === spot.cabinName)
    ),
    'every map hotspot should connect to a real cabin card'
  );
  console.assert(
    mapHotspots.every(
      (spot) => spot.x >= 0 && spot.x <= 100 && spot.y >= 0 && spot.y <= 100
    ),
    'map hotspot positions should use percentage coordinates'
  );
  console.assert(
    mapHotspots.every((spot) => spot.shortLabel && spot.shortLabel.length <= 2),
    'map pins should use tiny two-letter labels so they do not block the map'
  );
}

runDataTests();

function Button({ children, className = '', variant = 'solid', ...props }) {
  const base =
    'inline-flex items-center justify-center rounded-full px-5 py-3 text-sm font-bold transition focus:outline-none focus:ring-4 focus:ring-emerald-900/20';
  const variants = {
    solid: 'bg-emerald-900 text-white hover:bg-emerald-800',
    dark: 'bg-stone-950 text-white hover:bg-stone-800',
    light: 'bg-white text-emerald-950 hover:bg-emerald-50',
    outline:
      'border border-stone-300 bg-white/75 text-stone-950 hover:bg-white',
  };

  return (
    <button
      className={`${base} ${variants[variant] || variants.solid} ${className}`}
      {...props}
    >
      {children}
    </button>
  );
}
